
\subsection*{1.3 Landasan Hukum}
\addcontentsline{toc}{subsection}{1.3 Landasan Hukum}

\noindent Adapun landasan hukum dalam Pembuatan Pembuatan Profil PD dan 
Updating Data Bagian Organisasi adalah sebagai berikut:

\begin{enumerate}
    \item Undang-Undang Nomor 14 Tahun 2008 tentang Keterbukaan Informasi Publik;
    \item Undang-Undang Nomor 25 Tahun 2009 tentang Pelayanan Publik;
    \item Undang-Undang Nomor 11 Tahun 2008 tentang Informasi dan 
    Transaksi Elektronik (UU ITE);
    \item Undang-Undang Nomor 23 Tahun 2014 tentang Pemerintah Daerah;
    \item Peraturan Presiden Nomor 95 Tahun 2018 tentang Sistem Pemerintahan 
    berbasis Elektronik;
    \item Inpres No. 3 Tahun 2003 tentang “Kebijakan dan Strategi Nasional 
    Pengembangan e-Government”;
    \item Peraturan Menteri Dalam Negeri (Permendagri) Nomor 3 Tahun 2017 
    tentang Pedoman Pengelolaan Pelayanan Informasi dan Dokumentasi di 
    Lingkungan Kementerian Dalam Negeri dan Pemerintah Daerah;
    \item Peraturan Komisi Informasi Nomor 1 Tahun 2010 tentang Standar 
    Layanan Informasi Publik
    \item Peraturan Pemerintah Nomor 61 Tahun 2010 tentang Pelaksanaan 
    Undang-Undang Nomor 14 Tahun 2008 tentang Keterbukaan Informasi 
    Publik

\end{enumerate}
