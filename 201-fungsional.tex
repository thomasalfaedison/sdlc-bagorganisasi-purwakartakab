\subsubsection*{2.1.1 Landing Page}

\addcontentsline{toc}{subsubsubection}{2.1.1 Landing Page}

\noindent Berikut kebutuhan fungsional pada halaman Landing Page Aplikasi Bagorganisasi

\begin{longtable}{|c|p{5cm}|p{8cm}|}
    \hline
    \textbf{No} & \textbf{Fungsionalitas} & \textbf{Deskripsi} \\
    \hline
    1 & Pencarian Dokumen & Pengunjung dapat mencari dokumen publik berdasarkan kata kunci atau kategori. \\
    \hline
    2 & Akses ke Situs Eksternal & Pengunjung dapat mengakses tautan ke situs web resmi atau instansi lain. \\
    \hline
    3 & Unduh Dokumen & Pengunjung dapat mengunduh dokumen yang tersedia tanpa login. \\
    \hline
    4 & Melihat Berita & Menampilkan daftar berita atau pengumuman terbaru. \\
    \hline
    5 & Pencarian Berita & Pengunjung dapat mencari berita tertentu berdasarkan judul atau isi. \\
    \hline
    6 & Informasi Tentang Kami & Menyediakan informasi profil organisasi/instansi. \\
    \hline
    7 & Informasi Kontak & Menampilkan alamat, nomor telepon, email, atau peta lokasi. \\
    \hline
    8 & Melihat Pencapaian & Menampilkan data statistik atau prestasi instansi dalam bentuk teks/grafik. \\
    \hline
    \end{longtable}

\subsubsection*{2.1.2 Halaman Kelola Aplikasi Bagorganisasi}

\addcontentsline{toc}{subssubection}{2.1.2 Halaman Kelola Aplikasi Bagorganisasi}

\noindent Berikut kebutuhan fungsional pada halaman Kelola Aplikasi Bagorganisasi

\begin{longtable}{|c|p{4cm}|p{8cm}|}
    \hline
    \textbf{No} & \textbf{Requirement } & \textbf{Deskripsi} \\
    \hline
    1 & Login & Sistem harus menyediakan fitur login untuk pengguna agar dapat mengakses halaman kelola Aplikas Bagorganisasi. \\
    \hline
    2 & Input Berita & Admin dapat menambahkan berita mulai dari judul, gambar, status sorotan, dan status aktif. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    3 & Input Dokumen & Admin dapat menambahkan dokumen mulai dari kategori, nama, upload file, dan status publik. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    4 & Input Pencapaian & Admin dapat menambahkan data pencapaian mulai dari judul dan gambar. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    5 & Input Gambar Sampul & Admin dapat menambahkan gambar sampul mulai dari gambar dan status aktif. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    6 & Input Teks Depan & Admin dapat menambahkan teks depan mulai dari teks, sub teks, teks tombol, tautan tombol, dan status aktif. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    7 & Input Tentang Kami & Admin dapat menambahkan data tentang kami mulai dari Deskripsi, tautan/link, dan gambar. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    8 & Input Tautan Terkait & Admin dapat menambahkan data tautan terkait mulai dari nama, tautan/link, dan gambar. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    9 & Input Informasi & Admin dapat menambahkan data informasi mulai dari alamat, no telp/WA, email, dan map. Selain itu dapat mengubah dan menghapus data.\\
    \hline
    10 & Kelola User & Admin dapat menambahkan, mengubah, dan menghapus data user tertentu. \\
    \hline
\end{longtable}
