\subsection*{2.2 Kebutuhan Non-Fungsional}

\addcontentsline{toc}{subsection}{2.2 Kebutuhan Non-Fungsional}

\subsubsection*{2.2.1 Landing Page}

\addcontentsline{toc}{subsubsubection}{2.2.1 Landing Page}

\noindent Berikut kebutuhan non-fungsional pada halaman Landing Page Aplikasi Bagorganisasi

\begin{longtable}{|c|p{5cm}|p{8cm}|}
  \hline
  \textbf{No} & \textbf{Non-Fungsionalitas} & \textbf{Deskripsi} \\
  \hline
  1 & Responsif & Tampilan harus menyesuaikan untuk desktop, tablet, dan mobile. \\
  \hline
  2 & Performa Cepat & Halaman utama harus dimuat dalam waktu kurang dari 3 detik. \\
  \hline
  3 & Keamanan Dasar & Tidak ada celah keamanan yang membahayakan data pengunjung (misalnya XSS). \\
  \hline
  4 & Ketersediaan & Landing page harus dapat diakses minimal 99\% uptime. \\
  \hline
  \end{longtable}


\subsubsection*{2.2.2 Halaman Kelola Aplikasi Bagorganisasi}

\addcontentsline{toc}{subssubection}{2.2.2 Halaman Kelola Aplikasi Bagorganisasi}

\noindent Berikut kebutuhan non-fungsional dari halaman kelola Aplikasi Bagorganisasi

\begin{longtable}{|c|p{4cm}|p{8cm}|}
  \hline
    \textbf{No} & \textbf{Requirement } & \textbf{Deskripsi} \\
    \hline
      1 & Keamanan (Security) & Sistem harus mengenkripsi password dan membatasi akses berdasarkan peran pengguna. \\
      \hline
      2 & Ketersediaan (Availability) & Sistem harus dapat diakses minimal 98\% waktu kerja (uptime tinggi). \\
      \hline
      3 & Performa & Sistem harus dapat menampilkan data pencarian dalam waktu kurang dari 3 detik. \\
      \hline
      4 & Kemudahan Penggunaan (Usability) & Antarmuka pengguna harus ramah pengguna dan mudah dipahami oleh staf administrasi. \\
      \hline
\end{longtable}
